Vulnerability: Attacker-controlled targetOrigin in postMessage

modernFrame.aspx reads origin and channelId from URL hash fragment. These values are used as targetOrigin when sending postMessage to window.opener. An attacker page can window.open() the modernFrame with a crafted hash to receive sensitive data.

// modernframe.js - function g(a):
function g(a) {
  if (a.channelId && a.origin) return a;
  let b = window.location.hash;  // <-- reads from URL hash!
  let c = {channelId:"", origin:""};
  for (let a of b.slice(1).split("&")) {
    let [b,d] = a.split("=",3);
    c.hasOwnProperty(b) && (c[b] = decodeURIComponent(d));
  }
  return {...c, ...a};
}

// Then used as targetOrigin:
let {origin: h = location.origin} = g(config);
let m = window.parent !== window ? window.parent
      : ("*" !== h ? window.opener : undefined);   // uses window.opener!
m.postMessage({type:"initialize", channelId, replyTo:port}, h, [port]);
//                                                          ^^^ attacker-controlled!
